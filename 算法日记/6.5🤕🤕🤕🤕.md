### 核心思路
==贪心，只需要考虑第一个元素和它后一个元素会即可==
### 代码部分
```cpp
#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;

int n, x;
int a[100005];  // 开大一点，防止越界
long long sum = 0;  // 注意要用long long，可能超过int范围

int main() {
    cin >> n >> x;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    // 贪心：从左到右处理，每次只关心当前糖果盒和下一个糖果盒
    for (int i = 1; i < n; i++) {
        if (a[i] + a[i+1] > x) {
            int reduce = a[i] + a[i+1] - x;  // 需要减少的数量
            // 优先从后面的糖果盒中减少（贪心策略）
            if (a[i+1] >= reduce) {
                a[i+1] -= reduce;
            } else {
                a[i] -= (reduce - a[i+1]);
                a[i+1] = 0;
            }
            sum += reduce;
        }
    }
    
    cout << sum << endl;
    return 0;
}
```
### 时间复杂度
O(n)
