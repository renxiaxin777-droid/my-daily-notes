### 核心思想
==二分查找+前缀和==
### 代码部分
```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;
long long n, m, S;
long long w[MAXN], v[MAXN], L[MAXN], R[MAXN];
long long cnt[MAXN], sum[MAXN]; // 前缀和数组

// 计算参数为W时的检验结果Y
long long calculateY(long long W) {
    // 1. 预处理前缀和
    for (int i = 1; i <= n; ++i) {
        if (w[i] >= W) {
            cnt[i] = cnt[i - 1] + 1;
            sum[i] = sum[i - 1] + v[i];
        } else {
            cnt[i] = cnt[i - 1];
            sum[i] = sum[i - 1];
        }
    }

    // 2. 计算所有区间的Y值之和
    long long Y = 0;
    for (int i = 1; i <= m; ++i) {
        long long c = cnt[R[i]] - cnt[L[i] - 1];
        long long s = sum[R[i]] - sum[L[i] - 1];
        Y += c * s;
    }
    return Y;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n >> m >> S;
    long long maxW = 0, minW = 1e9;
    for (int i = 1; i <= n; ++i) {
        cin >> w[i] >> v[i];
        maxW = max(maxW, w[i]);
        minW = min(minW, w[i]);
    }
    for (int i = 1; i <= m; ++i) {
        cin >> L[i] >> R[i];
    }

    // 二分查找[left, right]区间
    long long left = minW - 1, right = maxW + 1;
    long long ans = 0x3f3f3f3f3f3f3f3f; // 初始化为极大值

    while (left <= right) {
        long long mid = left + (right - left) / 2;
        long long Y = calculateY(mid);
        long long diff = Y - S;
        
        ans = min(ans, abs(diff)); // 更新最小差值
        
        if (diff > 0) {
            left = mid + 1; // Y大于S，需增大W
        } else if (diff < 0) {
            right = mid - 1; // Y小于S，需减小W
        } else {
            ans = 0; // diff为0，找到最优解
            break;
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
### 复杂度
O(n+m)

```cpp
#include<iostream>
using namespace std;

int n;
struct hh {
    int timee;
    int qu;
} q[300050];  // 总乘客数可能很大，开大一点

int country[100050] = { 0 };  // 国家编号范围？题目说 ≤10^5

int main() {
    int ans = 0;
    cin >> n;
    
    int head = 0, tail = 0;  // ✅ head和tail定义在循环外面
    
    for (int i = 1; i <= n; i++) {
        int t, k;
        cin >> t >> k;
        
        // 1. 添加当前船的乘客
        while (k--) {
            int x;
            cin >> x;
            q[tail++] = {t, x};  // tail指向下一个空位置
            country[x]++;
            if (country[x] == 1)
                ans++;
        }
        
        // 2. 移除超过24小时的乘客（注意：head指向第一个有效元素）
        while (head < tail && t - q[head].timee >= 86400) {
            country[q[head].qu]--;
            if (country[q[head].qu] == 0)
                ans--;
            head++;  // ✅ 处理完当前head再移动
        }
        
        cout << ans << endl;
    }
    
    return 0;
}
```
### 复杂度
O(kn)

### 代码部分
```cpp
#include<iostream>
using namespace std;
int n;
int a[130][130];//存储题目中的矩阵
int sum[130][130];
int qz[130][130];//qz[i][j]指的是第i行到j的前缀和
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            cin>>a[i][j];
            qz[i][j]=qz[i][j-1]+a[i][j];//求前缀和
            sum[i][j]=qz[i][j]+sum[i-1][j];//计算sum
        }
    }
    int mx=-99999999;//存储答案
    for(int x1=1;x1<=n;x1++){
    	for(int y1=1;y1<=n;y1++){
    		for(int x2=1;x2<=n;x2++){
    			for(int y2=1;y2<=n;y2++){
    				if(x2<x1 || y2<y1) continue;//如果左上角比右下角还要大，就不用求了，下一个
    				mx=max(mx,sum[x2][y2]+sum[x1-1][y1-1]-sum[x2][y1-1]-sum[x1-1][y2]);//求最大值
    			}
    		}
    	}
    }
    cout<<mx;//输出
	return 0;
}
```
### 复杂度

```cpp
long long query(int x1, int y1, int x2, int y2) {
```
